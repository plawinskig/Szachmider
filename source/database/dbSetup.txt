PRAGMA foreign_keys = ON
CREATE TABLE Players(PId INTEGER PRIMARY KEY AUTOINCREMENT, PName TEXT UNIQUE NOT NULL)
CREATE TABLE Boards(BId INTEGER PRIMARY KEY AUTOINCREMENT, fileName TEXT UNIQUE NOT NULL)
CREATE TABLE Games(GId INTEGER PRIMARY KEY AUTOINCREMENT, PWhite INTEGER NOT NULL, PBlack INTEGER NOT NULL, playedBoard INTEGER NOT NULL, winner TEXT NOT NULL, date TEXT NOT NULL, FOREIGN KEY (PWhite) REFERENCES Players(PId) ON DELETE RESTRICT, FOREIGN KEY (PBlack) REFERENCES Players(PId) ON DELETE RESTRICT, FOREIGN KEY (playedBoard) REFERENCES Boards(BId) ON DELETE RESTRICT, CHECK(PBlack <> PWhite and winner in('B', 'W', 'D', 'N')))
CREATE TABLE MoveHistory(game INTEGER NOT NULL, turn INTEGER NOT NULL, pieceID TEXT NOT NULL, moveX INTEGER NOT NULL, moveY INTEGER NOT NULL, PRIMARY KEY (game, turn), FOREIGN KEY (game) REFERENCES Games(GId) ON DELETE CASCADE)
